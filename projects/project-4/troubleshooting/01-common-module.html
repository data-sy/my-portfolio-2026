# 직교좌표계 → 구면좌표계 변환
def to_spherical(x, y, z):
    r = sqrt(x**2 + y**2 + z**2)
    theta = arccos(z / r)
    phi = arctan2(y, x)
    return (r, theta, phi)

시도 2: 세 관절 각도 계산 공통 모듈 개발
# 공통 모듈: 세 관절의 각도를 반환
def calculate_angle(joint1, joint2, joint3):
    """
    joint2를 꼭짓점으로 하는 각도 계산
    예: calculate_angle(shoulder, elbow, wrist) → 팔꿈치 각도
    """
    vector1 = joint1 - joint2
    vector2 = joint3 - joint2

    cos_angle = dot(vector1, vector2) / (norm(vector1) * norm(vector2))
    angle = arccos(cos_angle)

    return degrees(angle)

# 이제 모든 운동에서 동일하게 사용
pushup_angle = calculate_angle(shoulder, elbow, wrist)
squat_angle = calculate_angle(hip, knee, ankle)
curl_angle = calculate_angle(shoulder, elbow, wrist)

  • 반복 패턴의 추상화: 반복되는 패턴을 발견하고 추상화하면 개발 생산성이 크게 향상됩니다. "비슷한 코드를 여러 번 작성하고 있다"는 느낌이 들면 공통화를 고민해볼 시점입니다.
  • 수학적 접근의 가치: 좌표계 변환이라는 수학적 개념이 실제 코드 설계에 큰 도움이 되었습니다. 문제의 본질을 이해하면 더 나은 해결책을 찾을 수 있습니다.
